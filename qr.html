APPS SCRIPT (with Drive support) 
function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);

    // ── Handle CSV save to Drive ──────────────────
    if (d.action === 'saveCSV') {
      var folder = DriveApp.getFolderById(d.folderId);
      var blob   = Utilities.newBlob(d.csv, 'text/csv', d.filename);
      var file   = folder.createFile(blob);
      return ContentService
        .createTextOutput(JSON.stringify({
          status:'ok',
          fileId: file.getId(),
          fileName: file.getName(),
          webViewLink: file.getUrl()
        }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ── Handle list files in folder ───────────────
    if (d.action === 'listFiles') {
      var folder = DriveApp.getFolderById(d.folderId);
      var files  = folder.getFilesByType('text/csv');
      var list   = [];
      while (files.hasNext()) {
        var f = files.next();
        list.push({
          id:       f.getId(),
          name:     f.getName(),
          size:     f.getSize(),
          modified: f.getLastUpdated().toLocaleString(),
          url:      f.getUrl()
        });
      }
      list.sort((a,b)=> b.modified.localeCompare(a.modified));
      return ContentService
        .createTextOutput(JSON.stringify({status:'ok', files: list}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ── Handle delete file ────────────────────────
    if (d.action === 'deleteFile') {
      var file = DriveApp.getFileById(d.fileId);
      file.setTrashed(true);
      return ContentService
        .createTextOutput(JSON.stringify({status:'ok'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ── Handle get file content (preview) ─────────
    if (d.action === 'getFile') {
      var file    = DriveApp.getFileById(d.fileId);
      var content = file.getBlob().getDataAsString('UTF-8');
      return ContentService
        .createTextOutput(JSON.stringify({status:'ok', content: content, name: file.getName()}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ── Default: append row to Sheet ─────────────
    var sheet = SpreadsheetApp
      .getActiveSpreadsheet()
      .getSheetByName('Scans');
    var no = Math.max(sheet.getLastRow(), 1);
    sheet.appendRow([no, d.name||'', d.qr||'', d.reason||'', d.time||new Date().toLocaleString()]);
    return ContentService
      .createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error', msg: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
